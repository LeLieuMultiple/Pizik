SPoschandler_server = "192.168.1.100" #adjust for your own setup
use_osc SPoschandler_server, 8000 #SPoschandler_Server on port 8000

live_loop :b0 do
  use_real_time
  n = sync "/osc/play0"
  puts n
  use_synth :tb303
  play 60,sustain: 1,release: 0.2, cutoff_attack: 0.5, cutoff_release: 0.5
end
